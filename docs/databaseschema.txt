Table user as U {
  name varchar [pk, increment]

}
Table lecture as L {
  id int [pk, increment] // auto-increment
  name varchar
  code varchar
  state varchar
  deleted enum
}

Table takepart as T {
  username varchar [ref: > U.name]
  lectid int [ref: > L.id]
  role enum
}

Table assignment as A {
  id int [pk, increment]
  name varchar
  type enum
  lectid int [ref: > L.id]
  duedate datetime
  status enum
  points int
  deleted enum
  properties text
}

Table submission as S {
  id int [pk, increment]
  date datetime
  assignid int [ref: > A.id]
  username varchar [ref: > U.name]
  commit_hash varchar
  auto_status enum
  manual_status enum
  score int
  properties text
  feedback_available boolean
}

Table group as G {
  username varchar [pk,ref: > U.name]
  lectid int [pk,ref: > L.id]
  name varchar
}
